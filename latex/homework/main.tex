\begin{algorithm}[H]
  \SetCommentSty{mycommfont}  % comment text typography
  \SetKwFunction{FSearch}{Search}
  \KwInput{Marked graph $G=(F,M,E)$\\ Targeting strategy $f: G \mapsto v \in (F \cup M)$}
  \KwOutput{List of transitions $T=(I_1, M_1),\ldots,(I_k,M_k)$}
  \BlankLine
  \tcc{choose the target vertex}
  $v := f(G)$\\

  \BlankLine
  Let $L$ be a list of the free vertices in the closed neighborhood of $v$\\
  Sort $L$ by free-degree in decreasing order\\
  Let $T$ be an empty list of transitions\\
  Let $M$ be an empty set\\

  \BlankLine
  \ForEach{$u$ in $L$}{
      Append pair $(\{u\}, M \setminus N(u))$ to list $T$\\
      Add $u$ to set $M$
  }

  \KwRet{T}
  \caption{Branch}
\end{algorithm}
